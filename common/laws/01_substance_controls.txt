law_no_substance_controls = {
	group = lawgroup_substance_controls
	
	icon = "gfx/interface/icons/law_icons/no_substance_controls.dds"
	
	progressiveness = 0
	
	on_enact = {
		recalculate_pop_ig_support = yes
	}
	
	on_activate = {
		if = {
			limit = {
				is_banning_goods = g:liquor
			}
			remove_banned_goods = g:liquor
			remove_modifier = liqour_ban_authority
		}
		if = {
			limit = {
				is_banning_goods = g:opium
			}
			remove_banned_goods = g:opium
			remove_modifier = opium_ban_authority
		}
	}

	modifier = {
		
	}
	
	possible_political_movements = {
		law_opium_ban
		law_prohibition
		law_regulated_alcohol_and_tobacco
	}
	
	pop_support = {
		value = 0
	}

	can_impose = { 
		is_subject_of = scope:initiator	
	}

	ai_will_do = {
		exists = ruler
		ruler = {
			has_ideology = ideology:ideology_anarchist
		}
	}

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = { 
				has_journal_entry = je_government_petition
				has_variable = desired_law_var
                scope:law = var:desired_law_var
			}
			add = 750
		}
	}
}

	unlocking_technologies = {
		mass_surveillance
	}
	
	on_enact = {
		recalculate_pop_ig_support = yes
	}
	
	on_activate = {
		if = {
			limit = {
				is_banning_goods = g:liquor
			}
			remove_banned_goods = g:liquor
			remove_modifier = liqour_ban_authority
			add_taxed_goods = g:liquor

		desc = "POP_BUREAUCRATS"
			if = {
				limit = { 
					is_pop_type = bureaucrats
				}
				value = 0.25
			}
	}

	can_impose = { 
		is_subject_of = scope:initiator	
	} 

	ai_enact_weight_modifier = { #Petitions
		value = 0
		
		if = {
			limit = {

				has_variable = desired_law_var
