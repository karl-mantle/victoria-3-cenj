# Die Groot Trek

je_the_great_trek = {
	
	icon = "gfx/interface/icons/event_icons/event_scales.dds"
	
	is_shown_when_inactive = {
		exists = c:SAF
		c:SAF = ROOT
	}

	possible = {
		exists = c:SAF
		c:SAF = ROOT
	}

	immediate = {
		set_variable = {
			name = great_trek_var
			value = 0
		}
	}

	on_monthly_pulse = {
		# Zulu conflict ?
		# Accept Boers demand? change some sort of law?
		effect = {
			if = {
				limit = { 
					AND = {
						OR = { exists = c:WIB exists = c:ZOU }
						NOT = { has_variable = great_trek_started }
					}
				}
				set_variable = great_trek_started
				change_variable = {
					name = great_trek_var
					add = 2
				}
			}
		}
	}

	complete = {
		scope:journal_entry = {
			is_goal_complete = yes
		}
	}

	current_value = {
		value = var:great_trek_var
	}

	goal_add_value = {
		add = 4
	}

	progressbar = yes

	on_complete = {
		trigger_event = {
			id = boer_republics.1
		}
	}

	should_be_pinned_by_default = yes

}

# Natal Colony

je_natal_colony = {
	
	icon = "gfx/interface/icons/event_icons/event_scales.dds"
	
	is_shown_when_inactive = {
		AND = {
			exists = c:NAL
			c:NAL = ROOT
		}
	}

	possible = {
		year >= 1843
		exists = c:NAL
		# cordial relations between natalia and the SAF
	}

	on_monthly_pulse = {
		effect = {
			if = {
				limit = { game_date >= 1843.7.30 }
				trigger_event = {
					id = boer_republics.4
				}
			}
		}
	}

	complete = {
		OR = { 
			c:NAL = { is_subject_of = c:GBR }
			c:NAL = { is_subject_of = c:SAF }
		}
	}

	on_complete = {
		c:NAL = {
			remove_modifier = volksraad_government
			add_primary_culture = cu:british
		}
		# Trigger Event where Boers leave Natal to go to Oranje and Transvaal
	}

	should_be_pinned_by_default = yes
}
